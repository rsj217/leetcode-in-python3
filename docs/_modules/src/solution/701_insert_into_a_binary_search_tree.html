        """
        :type root: TreeNode

        """
        return self.insert_dfs(root, val)



[docs]    def insert_dfs(self, node, val):
        if node is None:
            return TreeNode(val)
        if val < node.val:
            node.left = self.insert_dfs(node.left, val)
        elif node.val < val:
            node.right = self.insert_dfs(node.right, val)
        return node

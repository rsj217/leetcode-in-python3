        if node is None:
            return node
        return self._minimum(node)

    def _insert(self, node, val):
        """ 以 ``node`` 为 ``root`` 的二叉搜索树种插入 ``val``，并返回新的 ``root`` """
        if node is None:
            return BSTreeNode(val=val)
        elif val < node.val:
            node.left = self._insert(node.left, val)
        else:  # node.val < val
            node.right = self._insert(node.right, val)
        return node
